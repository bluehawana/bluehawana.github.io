function calculateBilling(duration, serviceType) {
  const rates = {
    meeting: 15, // 1-on-1 meetings
    consulting: 25, // All other services
    development: 25,
    bridging: 25,
    infrastructure: 25,
    monitoring: 25,
    containers: 25,
    devops: 25
  };
  return Math.ceil(duration) * rates[serviceType];
}
