// Safety-first voice command processing
class SafetyVoiceProcessor {
    processCommand(command: string): CommandResult {
        // Validate safety constraints before execution
        if (this.isVehicleInMotion() && this.requiresVisualAttention(command)) {
            return this.deferCommand(command);
        }
        return this.executeCommand(command);
    }
}
