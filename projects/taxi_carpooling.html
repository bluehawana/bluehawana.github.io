// Advanced matching algorithm with ML optimization
@Component
public class MLMatchingEngine {
    
    public List<OptimalGroup> findOptimalGrouping(
        List<RideRequest> requests, 
        LocationMatrix matrix
    ) {
        // Apply K-means clustering for geographical grouping
        KMeansResult clusters = kMeansAlgorithm.cluster(
            requests.stream()
                .map(r -> new GeoPoint(r.getLat(), r.getLng()))
                .collect(toList())
        );
        
        // Optimize routes within each cluster
        return clusters.getClusters()
            .parallelStream()
            .map(cluster -> routeOptimizer.optimize(cluster))
            .collect(toList());
    }
}
